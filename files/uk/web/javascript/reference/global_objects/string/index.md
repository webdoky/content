## –û–ø–∏—Å

–†—è–¥–∫–∏ –∫–æ—Ä–∏—Å–Ω—ñ –¥–ª—è –∑–±–µ—Ä—ñ–≥–∞–Ω–Ω—è —Ç–∏—Ö –¥–∞–Ω–∏—Ö, —è–∫—ñ –º–æ–∂–Ω–∞ –ø—Ä–µ–¥—Å—Ç–∞–≤–∏—Ç–∏ –≤ —Ç–µ–∫—Å—Ç–æ–≤—ñ–π —Ñ–æ—Ä–º—ñ. –°–µ—Ä–µ–¥ –Ω–∞–π—É–∂–∏–≤–∞–Ω—ñ—à–∏—Ö –æ–ø–µ—Ä–∞—Ü—ñ–π –∑ —Ä—è–¥–∫–∞–º–∏: –≤–∏–∑–Ω–∞—á–µ–Ω–Ω—è —ó—Ö–Ω—å–æ—ó {{jsxref("String/length", "–¥–æ–≤–∂–∏–Ω–∏")}}, –∑–±–∏—Ä–∞–Ω–Ω—è —ñ –∑—á–µ–ø–ª–µ–Ω–Ω—è —ó—Ö –¥–æ–∫—É–ø–∏ –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é [—Ä—è–¥–∫–æ–≤–∏—Ö –æ–ø–µ—Ä–∞—Ç–æ—Ä—ñ–≤ `+` —Ç–∞ `+=`](/uk/docs/Web/JavaScript/Guide/Expressions_and_operators#riadkovi-operatory), –ø–µ—Ä–µ–≤—ñ—Ä–∫—É –Ω–∞—è–≤–Ω–æ—Å—Ç—ñ —á–∏ –∑–Ω–∞—Ö–æ–¥–∂–µ–Ω–Ω—è –ø–æ–∑–∏—Ü—ñ—ó –ø—ñ–¥—Ä—è–¥–∫–∞ –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é –º–µ—Ç–æ–¥—É {{jsxref("String/indexOf", "indexOf()")}} —Ç–∞ –≤–∏—Ç—è–≥–∞–Ω–Ω—è –ø–µ–≤–Ω–∏—Ö –ø—ñ–¥—Ä—è–¥–∫—ñ–≤ –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é –º–µ—Ç–æ–¥—É {{jsxref("String/substring", "substring()")}}.

### –°—Ç–≤–æ—Ä–µ–Ω–Ω—è —Ä—è–¥–∫—ñ–≤

–†—è–¥–∫–∏ –º–æ–∂–Ω–∞ —Å—Ç–≤–æ—Ä—é–≤–∞—Ç–∏ —è–∫ –ø—Ä–∏–º—ñ—Ç–∏–≤–∏, –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—é—á–∏ —Ä—è–¥–∫–æ–≤—ñ –ª—ñ—Ç–µ—Ä–∞–ª–∏; –∞–±–æ –∂ —è–∫ –æ–±'—î–∫—Ç–∏ ‚Äì –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é –∫–æ–Ω—Å—Ç—Ä—É–∫—Ç–æ—Ä–∞ {{jsxref("String/String", "String()")}}:

```js

const string1 = "–†—è–¥–∫–æ–≤–∏–π –ø—Ä–∏–º—ñ—Ç–∏–≤";
const string2 = '–¢–∞–∫–æ–∂ —Ä—è–¥–∫–æ–≤–∏–π –ø—Ä–∏–º—ñ—Ç–∏–≤';
const string3 = `–Ü —â–µ –æ–¥–∏–Ω —Ä—è–¥–∫–æ–≤–∏–π –ø—Ä–∏–º—ñ—Ç–∏–≤`;
```

```js
const string4 = new String("–û–±'—î–∫—Ç —Ä—è–¥–∫–∞");
```

–†—è–¥–∫–æ–≤—ñ –ø—Ä–∏–º—ñ—Ç–∏–≤–∏ —Ç–∞ –æ–±'—î–∫—Ç–∏-—Ä—è–¥–∫–∏ –ø–æ–¥—ñ–ª—è—é—Ç—å –∑–Ω–∞—á–Ω—É —á–∞—Å—Ç–∏–Ω—É —Å–≤–æ—î—ó –ø–æ–≤–µ–¥—ñ–Ω–∫–∏, –∞–ª–µ –º–∞—é—Ç—å –ø–µ–≤–Ω—ñ –≤–∞–∂–ª–∏–≤—ñ –≤—ñ–¥–º—ñ–Ω–Ω–æ—Å—Ç—ñ —Ç–∞ –∫–∞–≤–µ—Ä–∑–∏. –î–∏–≤—ñ—Ç—å—Å—è —Ä–æ–∑–¥—ñ–ª "[–†—è–¥–∫–∏-–ø—Ä–∏–º—ñ—Ç–∏–≤–∏ —Ç–∞ —Ä—è–¥–∫–∏-–æ–±'—î–∫—Ç–∏](#riadky-prymityvy-i-riadky-obiekty)" –Ω–∏–∂—á–µ.

–†—è–¥–∫–æ–≤—ñ –ª—ñ—Ç–µ—Ä–∞–ª–∏ –ø–æ–∑–Ω–∞—á–∞—é—Ç—å—Å—è –æ–¥–∏–Ω–∞—Ä–Ω–∏–º–∏ –∞–±–æ –ø–æ–¥–≤—ñ–π–Ω–∏–º–∏ –ª–∞–ø–∫–∞–º–∏ ‚Äî —è–∫—ñ –≤ —Å–µ—Ä–µ–¥–æ–≤–∏—â—ñ JavaScript –æ–ø—Ä–∞—Ü—å–æ–≤—É—é—Ç—å—Å—è —ñ–¥–µ–Ω—Ç–∏—á–Ω–æ ‚Äî –∞–±–æ –∂ —á–µ—Ä–µ–∑ —Å–∏–º–≤–æ–ª ¬´–≥—Ä–∞–≤—ñ—Å¬ª <kbd>`</kbd>. –û—Å—Ç–∞–Ω–Ω—ñ–π –ø–æ–∑–Ω–∞—á–∞—î —Ç–∞–∫ –∑–≤–∞–Ω–∏–π [—à–∞–±–ª–æ–Ω–Ω–∏–π –ª—ñ—Ç–µ—Ä–∞–ª](/uk/docs/Web/JavaScript/Reference/Template_literals) ‚Äî —Ñ–æ—Ä–º—É —Ä—è–¥–∫–æ–≤–æ–≥–æ –ª—ñ—Ç–µ—Ä–∞–ª–∞, —è–∫–∞ –¥–∞—î –∑–º–æ–≥—É –≤—Å—Ç–∞–≤–ª—è—Ç–∏ –≤–∏—Ä–∞–∑–∏ –≤ —Ä—è–¥–æ–∫ (—ñ–Ω—Ç–µ—Ä–ø–æ–ª—è—Ü—ñ—è). –ë—ñ–ª—å—à–µ —ñ–Ω—Ñ–æ—Ä–º–∞—Ü—ñ—ó –ø—Ä–æ –∑–∞–ø–∏—Å —Ä—è–¥–∫–æ–≤–∏—Ö –ª—ñ—Ç–µ—Ä–∞–ª—ñ–≤ ‚Äì —É [–ª–µ–∫—Å–∏—á–Ω—ñ–π –≥—Ä–∞–º–∞—Ç–∏—Ü—ñ](/uk/docs/Web/JavaScript/Reference/Lexical_grammar#riadkovi-literaly).

### –î–æ—Å—Ç—É–ø –¥–æ –æ–∫—Ä–µ–º–∏—Ö —Å–∏–º–≤–æ–ª—ñ–≤

–Ü—Å–Ω—É—é—Ç—å –¥–≤–∞ —Å–ø–æ—Å–æ–±–∏ –¥–æ—Å—Ç—É–ø–∏—Ç–∏—Å—è –¥–æ –æ–∫—Ä–µ–º–æ–≥–æ —Å–∏–º–≤–æ–ª—É —Ä—è–¥–∫–∞. –ü–µ—Ä—à–∏–π ‚Äî —Ü–µ –º–µ—Ç–æ–¥ {{jsxref("String/charAt", "charAt()")}}:

```js
"cat".charAt(1); // –¥–∞—î –∑–Ω–∞—á–µ–Ω–Ω—è "a"
```

–Ü–Ω—à–∏–π —Å–ø–æ—Å—ñ–± (—è–∫–∏–π –±—É–ª–æ –¥–æ–¥–∞–Ω–æ –≤ ECMAScript 5) –ø–æ–ª—è–≥–∞—î —É —Ä–æ–∑–≥–ª—è–¥—ñ —Ä—è–¥–∫–∞ —è–∫ –º–∞—Å–∏–≤–æ–ø–æ–¥—ñ–±–Ω–æ–≥–æ –æ–±'—î–∫—Ç–∞, –¥–µ –∫–æ–∂–Ω–∏–π —Å–∏–º–≤–æ–ª –≤—ñ–¥–ø–æ–≤—ñ–¥–∞—î –ø–µ–≤–Ω–æ–º—É —á–∏—Å–ª–æ–≤–æ–º—É —ñ–Ω–¥–µ–∫—Å—É:

```js
"cat"[1]; // –¥–∞—î –∑–Ω–∞—á–µ–Ω–Ω—è "a"
```

–°–ø—Ä–æ–±–∞ –≤–∏–¥–∞–ª–∏—Ç–∏ —Å–∏–º–≤–æ–ª –∞–±–æ –ø—Ä–∏—Å–≤–æ—ó—Ç–∏ –π–æ–º—É –∑–Ω–∞—á–µ–Ω–Ω—è –ø—ñ–¥ —á–∞—Å –∑–≤–µ—Ä—Ç–∞–Ω–Ω—è –¥–æ —Å–∏–º–≤–æ–ª—É –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é –∫–≤–∞–¥—Ä–∞—Ç–Ω–∏—Ö –¥—É–∂–æ–∫ ‚Äì –Ω–µ —Å–ø—Ä–∞—Ü—é—î. –í–ª–∞—Å—Ç–∏–≤–æ—Å—Ç—ñ —Ä—è–¥–∫–∞, –¥–æ —è–∫–∏—Ö –≤ —Ü—å–æ–º—É –≤–∏–ø–∞–¥–∫—É –æ—Ç—Ä–∏–º—É—î –¥–æ—Å—Ç—É–ø –ø—Ä–æ–≥—Ä–∞–º–∞, –Ω–µ –¥–æ—Å—Ç—É–ø–Ω—ñ –Ω—ñ –¥–ª—è –∑–∞–ø–∏—Å—É, –Ω—ñ –¥–ª—è –Ω–∞–ª–∞—à—Ç—É–≤–∞–Ω–Ω—è. (–î–æ–∫–ª–∞–¥–Ω—ñ—à–µ –ø—Ä–æ —Ü–µ –Ω–∞ —Å—Ç–æ—Ä—ñ–Ω—Ü—ñ {{jsxref("Object.defineProperty()")}}.)

### –ü–æ—Ä—ñ–≤–Ω—è–Ω–Ω—è —Ä—è–¥–∫—ñ–≤

–î–ª—è –ø–æ—Ä—ñ–≤–Ω—è–Ω–Ω—è —Ä—è–¥–∫—ñ–≤ —Å–ª—ñ–¥ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–≤–∞—Ç–∏ [–æ–ø–µ—Ä–∞—Ç–æ—Ä–∏ –±—ñ–ª—å—à–µ —Ç–∞ –º–µ–Ω—à–µ](/uk/docs/Web/JavaScript/Reference/Operators):

```js
const a = "a";
const b = "b";
if (a < b) {
  // true
  console.log(`${a} –º–µ–Ω—à–µ –∑–∞ ${b}`);
} else if (a > b) {
  console.log(`${a} –±—ñ–ª—å—à–µ –∑–∞ ${b}`);
} else {
  console.log(`${a} —Ç–∞ ${b} ‚Äì —Ä—ñ–≤–Ω—ñ.`);
}
```

–ó–≤–µ—Ä–Ω—ñ—Ç—å —É–≤–∞–≥—É, —â–æ –≤—Å—ñ –æ–ø–µ—Ä–∞—Ç–æ—Ä–∏ –ø–æ—Ä—ñ–≤–Ω—è–Ω–Ω—è, –≤–∫–ª—é—á–Ω–æ –∑ [`===`](/uk/docs/Web/JavaScript/Reference/Operators/Strict_equality) —ñ [`==`](/uk/docs/Web/JavaScript/Reference/Operators/Equality), –ø–æ—Ä—ñ–≤–Ω—é—é—Ç—å —Ä—è–¥–∫–∏ –∑ —É—Ä–∞—Ö—É–≤–∞–Ω–Ω—è–º —Ä–µ–≥—ñ—Å—Ç—Ä—É. –ó–∞–≥–∞–ª—å–Ω–æ–ø—Ä–∏–π–Ω—è—Ç–Ω–∏–π —Å–ø–æ—Å—ñ–± –ø–æ—Ä—ñ–≤–Ω—é–≤–∞—Ç–∏ —Ä—è–¥–∫–∏ –±–µ–∑ —É—Ä–∞—Ö—É–≤–∞–Ω–Ω—è —Ä–µ–≥—ñ—Å—Ç—Ä—É ‚Äì –ø–µ—Ä–µ–≤–æ–¥–∏—Ç–∏ –æ–±–∏–¥–≤–∞, –ø–µ—Ä–µ–¥ —ó—Ö –ø–æ—Ä—ñ–≤–Ω—è–Ω–Ω—è–º, –¥–æ –æ–¥–Ω–æ–≥–æ —Ä–µ–≥—ñ—Å—Ç—Ä—É (–≤–∏—â–æ–≥–æ —á–∏ –Ω–∏–∂—á–æ–≥–æ).

```js
function areEqualCaseInsensitive(str1, str2) {
  return str1.toUpperCase() === str2.toUpperCase();
}
```

–í–∏–±—ñ—Ä —Ç–æ–≥–æ, —á–∏ –ø–µ—Ä–µ—Ç–≤–æ—Ä—é–≤–∞—Ç–∏ –∑ [`toUpperCase()`](/uk/docs/Web/JavaScript/Reference/Global_Objects/String/toUpperCase), —á–∏ –∑ [`toLowerCase()`](/uk/docs/Web/JavaScript/Reference/Global_Objects/String/toLowerCase), —î –∑–¥–µ–±—ñ–ª—å—à–æ–≥–æ –¥–æ–≤—ñ–ª—å–Ω–∏–º, —ñ –∂–æ–¥–µ–Ω –∑ –¥–≤–æ—Ö –≤–∞—Ä—ñ–∞–Ω—Ç—ñ–≤ –Ω–µ –º–∞—î —Ü—ñ–ª–∫–æ–≤–∏—Ç–æ—ó –Ω–∞–¥—ñ–π–Ω–æ—Å—Ç—ñ –ø–æ–∑–∞ –º–µ–∂–∞–º–∏ –ª–∞—Ç–∏–Ω—Å—å–∫–æ–≥–æ –∞–ª—Ñ–∞–≤—ñ—Ç—É. –ù–∞–ø—Ä–∏–∫–ª–∞–¥, —ñ –Ω—ñ–º–µ—Ü—å–∫—É –ª—ñ—Ç–µ—Ä—É `√ü`, —ñ –ª—ñ—Ç–µ—Ä–∏ `ss` `toUpperCase()` –ø–µ—Ä–µ—Ç–≤–æ—Ä—é—î –Ω–∞ `SS`, —Ç–∏–º —á–∞—Å–æ–º —Ç—É—Ä–µ—Ü—å–∫–∞ –ª—ñ—Ç–µ—Ä–∞ `ƒ±` –º–æ–∂–µ –ø–æ–º–∏–ª–∫–æ–≤–æ –±—É—Ç–∏ –∑–∞—è–≤–ª–µ–Ω–∞ `toLowerCase()` —è–∫ –Ω–µ—Ä—ñ–≤–Ω–∞ `I`, —è–∫—â–æ —è–≤–Ω–æ –Ω–µ –≤–∏–∫–æ—Ä–∏—Å—Ç–∞—Ç–∏ [`toLocaleLowerCase("tr")`](/uk/docs/Web/JavaScript/Reference/Global_Objects/String/toLocaleLowerCase).

```js
const areEqualInUpperCase = (str1, str2) =>
  str1.toUpperCase() === str2.toUpperCase();
const areEqualInLowerCase = (str1, str2) =>
  str1.toLowerCase() === str2.toLowerCase();
areEqualInUpperCase("√ü", "ss"); // true; –ø–æ–≤–∏–Ω–Ω–æ –±—É—Ç–∏ false
areEqualInLowerCase("ƒ±", "I"); // false; –ø–æ–≤–∏–Ω–Ω–æ –±—É—Ç–∏ true
```

–ù–∞–¥—ñ–π–Ω–µ —Ä—ñ—à–µ–Ω–Ω—è –∑ —É—Ä–∞—Ö—É–≤–∞–Ω–Ω—è–º –ª–æ–∫–∞–ª—ñ –¥–ª—è –ø–µ—Ä–µ–≤—ñ—Ä–∫–∏ –Ω–µ—á—É—Ç–ª–∏–≤–æ—ó –¥–æ —Ä–µ–≥—ñ—Å—Ç—Ä—É —Ä—ñ–≤–Ω–æ—Å—Ç—ñ ‚Äì –∑–∞—Å—Ç–æ—Å–æ–≤—É–≤–∞—Ç–∏ API {{jsxref("Intl.Collator")}} —á–∏ –º–µ—Ç–æ–¥ —Ä—è–¥–∫–∞ [`localeCompare()`](/uk/docs/Web/JavaScript/Reference/Global_Objects/String/localeCompare), ‚Äì –≤–æ–Ω–∏ –ø–æ–¥—ñ–ª—è—é—Ç—å –æ–¥–Ω–∞–∫–æ–≤–∏–π —ñ–Ω—Ç–µ—Ä—Ñ–µ–π—Å, ‚Äì –∑—ñ –∑–Ω–∞—á–µ–Ω–Ω—è–º –æ–ø—Ü—ñ—ó [`sensitivity`](/uk/docs/Web/JavaScript/Reference/Global_Objects/Intl/Collator/Collator#sensitivity) `"accent"` —á–∏ `"base"`.

```js

  str1.localeCompare(str2, locale, { sensitivity: "accent" }) === 0;
areEqual("√ü", "ss", "de"); // false
areEqual("ƒ±", "I", "tr"); // true
```

–ú–µ—Ç–æ–¥ `localeCompare()` –¥–∞—î –∑–º–æ–≥—É –ø–æ—Ä—ñ–≤–Ω—é–≤–∞—Ç–∏ —Ä—è–¥–∫–∏ —É –ø–æ–¥—ñ–±–Ω–∏–π –¥–æ `strcmp()` —Å–ø–æ—Å—ñ–±: –¥–∞—î –∑–º–æ–≥—É —Å–æ—Ä—Ç—É–≤–∞—Ç–∏ —Ä—è–¥–∫–∏ –∑ —É—Ä–∞—Ö—É–≤–∞–Ω–Ω—è–º –ª–æ–∫–∞–ª—ñ.

### –†—è–¥–∫–∏-–ø—Ä–∏–º—ñ—Ç–∏–≤–∏ —ñ —Ä—è–¥–∫–∏-–æ–±'—î–∫—Ç–∏

–ó–∞—É–≤–∞–∂—Ç–µ, —â–æ JavaScript —Ä–æ–∑—Ä—ñ–∑–Ω—è—î –æ–±'—î–∫—Ç–∏ `String` —Ç–∞ {{Glossary("Primitive", "–ø—Ä–∏–º—ñ—Ç–∏–≤–Ω—ñ –∑–Ω–∞—á–µ–Ω–Ω—è —Ä—è–¥–∫—ñ–≤")}}. (–¶–µ —Å–ø—Ä–∞–≤–µ–¥–ª–∏–≤–æ —Ç–∞–∫–æ–∂ –¥–ª—è {{jsxref("Boolean", "–±—É–ª–µ–≤–∏—Ö –∑–Ω–∞—á–µ–Ω—å")}} —ñ {{jsxref("Number", "—á–∏—Å–µ–ª")}}.)

–†—è–¥–∫–æ–≤—ñ –ª—ñ—Ç–µ—Ä–∞–ª–∏ (–≤–∏–¥—ñ–ª–µ–Ω—ñ –æ–¥–∏–Ω–∞—Ä–Ω–∏–º–∏ –∞–±–æ –ø–æ–¥–≤—ñ–π–Ω–∏–º–∏ –ª–∞–ø–∫–∞–º–∏), –∞ —Ç–∞–∫–æ–∂ —Ä—è–¥–∫–∏, –ø–æ–≤–µ—Ä–Ω—É—Ç—ñ –∑ –≤–∏–∫–ª–∏–∫—ñ–≤ `String` –±–µ–∑ –∫–æ–Ω—Ç–µ–∫—Å—Ç—É –∫–æ–Ω—Å—Ç—Ä—É–∫—Ç–æ—Ä–∞ (—Ç–æ–±—Ç–æ –≤–∏–∫–ª–∏–∫—ñ–≤, –≤–∏–∫–æ–Ω–∞–Ω–∏—Ö –±–µ–∑ –∫–ª—é—á–æ–≤–æ–≥–æ —Å–ª–æ–≤–∞ {{jsxref("Operators/new", "new")}}), —î —Ä—è–¥–∫–∞–º–∏-–ø—Ä–∏–º—ñ—Ç–∏–≤–∞–º–∏. –ö–æ–ª–∏ –≤—ñ–¥–±—É–≤–∞—î—Ç—å—Å—è —Å–ø—Ä–æ–±–∞ –≤–∏–∫–ª–∏–∫–∞—Ç–∏ –º–µ—Ç–æ–¥ —á–∏ –∑–≤–µ—Ä–Ω—É—Ç–∏—Å—è –¥–æ –≤–ª–∞—Å—Ç–∏–≤–æ—Å—Ç—ñ –ø—Ä–∏–º—ñ—Ç–∏–≤–Ω–æ–≥–æ —Ä—è–¥–∫–∞, JavaScript –∞–≤—Ç–æ–º–∞—Ç–∏—á–Ω–æ –æ–±–≥–æ—Ä—Ç–∞—î –ø—Ä–∏–º—ñ—Ç–∏–≤ —É –≤–∏–∫–ª–∏–∫ –∫–æ–Ω—Å—Ç—Ä—É–∫—Ç–æ—Ä–∞, —ñ –≤–∂–µ –ø–æ—Ç—ñ–º ‚Äì –Ω–∞ –æ–±'—î–∫—Ç—ñ-–æ–±–≥–æ—Ä—Ç—Ü—ñ –∑–≤–µ—Ä—Ç–∞—î—Ç—å—Å—è –¥–æ –º–µ—Ç–æ–¥—É —á–∏ –≤–ª–∞—Å—Ç–∏–≤–æ—Å—Ç—ñ.

```js

```

> **–ó–∞—Å—Ç–µ—Ä–µ–∂–µ–Ω–Ω—è:** –°–ª—ñ–¥ —É—Ç—Ä–∏–º—É–≤–∞—Ç–∏—Å—è –≤—ñ–¥ –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–Ω—è `String` —è–∫ –∫–æ–Ω—Å—Ç—Ä—É–∫—Ç–æ—Ä–∞.

–†—è–¥–∫–æ–≤—ñ –ø—Ä–∏–º—ñ—Ç–∏–≤–∏ —Ç–∞ –æ–±'—î–∫—Ç–∏ —Ç–∏–ø—É `String` —Ç–∞–∫–æ–∂ –≤–∏–¥–∞—é—Ç—å —Ä—ñ–∑–Ω—ñ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∏ –ø—ñ–¥ —á–∞—Å –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–Ω—è {{jsxref("Global_Objects/eval", "eval()")}}. –ü—Ä–∏–º—ñ—Ç–∏–≤–∏, –ø–µ—Ä–µ–¥–∞–Ω—ñ –¥–æ `eval`, –≤–≤–∞–∂–∞—é—Ç—å—Å—è –≤–∏—Ö—ñ–¥–Ω–∏–º –∫–æ–¥–æ–º; –Ω–∞—Ç–æ–º—ñ—Å—Ç—å –æ–±'—î–∫—Ç–∏ `String` –æ–ø—Ä–∞—Ü—å–æ–≤—É—é—Ç—å—Å—è —è–∫ –≤—Å—ñ —ñ–Ω—à—ñ –æ–±'—î–∫—Ç–∏, —ñ–∑ –ø–æ–≤–µ—Ä–Ω–µ–Ω–Ω—è–º –æ–±'—î–∫—Ç–∞ –≤ —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ñ. –ù–∞–ø—Ä–∏–∫–ª–∞–¥:

```js
const s1 = "2 + 2"; // —Å—Ç–≤–æ—Ä—é—î —Ä—è–¥–∫–æ–≤–∏–π –ø—Ä–∏–º—ñ—Ç–∏–≤

```

–¢–∞–∫–∏–º —á–∏–Ω–æ–º, –∫–æ–¥ –º–æ–∂–µ –ª–∞–º–∞—Ç–∏—Å—å –≤ –º—ñ—Å—Ü—è—Ö, –¥–µ –æ—á—ñ–∫—É—î—Ç—å—Å—è –ø—Ä–∏–º—ñ—Ç–∏–≤–Ω–∏–π —Ä—è–¥–æ–∫, –∞–ª–µ –Ω–∞—Ç–æ–º—ñ—Å—Ç—å –ø—Ä–∏—Ö–æ–¥–∏—Ç—å –æ–±'—î–∫—Ç `String`. –•–æ—á–∞ —É –±—ñ–ª—å—à–æ—Å—Ç—ñ –≤–∏–ø–∞–¥–∫—ñ–≤ –ø—Ä–æ —Ü—é —Ä—ñ–∑–Ω–∏—Ü—é —Ö–≤–∏–ª—é–≤–∞—Ç–∏—Å—å –Ω–µ –≤–∞—Ä—Ç–æ.

–û–±'—î–∫—Ç `String` –º–æ–∂–Ω–∞ –∑–∞–≤–∂–¥–∏ –ø–µ—Ä–µ—Ç–≤–æ—Ä–∏—Ç–∏ –Ω–∞ –π–æ–≥–æ –ø—Ä–∏–º—ñ—Ç–∏–≤–Ω–∏–π –≤—ñ–¥–ø–æ–≤—ñ–¥–Ω–∏–∫, –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–≤—à–∏ –º–µ—Ç–æ–¥ {{jsxref("String/valueOf", "valueOf()")}}.

```js
console.log(eval(s2.valueOf())); // –ø–æ–≤–µ—Ä—Ç–∞—î —á–∏—Å–ª–æ 4
```

### –ó–≤–µ–¥–µ–Ω–Ω—è –¥–æ —Ä—è–¥–∫–∞

–ß–∏–º–∞–ª–æ –≤–±—É–¥–æ–≤–∞–Ω–∏—Ö –æ–ø–µ—Ä–∞—Ü—ñ–π, –∫–æ—Ç—Ä—ñ –æ—á—ñ–∫—É—é—Ç—å –Ω–∞ —Ä—è–¥–∫–∏, —Å–ø–µ—Ä—à—É –∑–≤–æ–¥—è—Ç—å —Å–≤–æ—ó –∞—Ä–≥—É–º–µ–Ω—Ç–∏ –¥–æ —Ç–∏–ø—É —Ä—è–¥–∫–∞ (–∑–¥–µ–±—ñ–ª—å—à–æ–≥–æ —Å–∞–º–µ —á–µ—Ä–µ–∑ —Ü–µ –æ–±'—î–∫—Ç–∏ `String` –ø–æ–≤–æ–¥—è—Ç—å—Å—è –ø–æ–¥—ñ–±–Ω–æ –¥–æ —Ä—è–¥–∫–æ–≤–∏—Ö –ø—Ä–∏–º—ñ—Ç–∏–≤—ñ–≤). [–¶—è –æ–ø–µ—Ä–∞—Ü—ñ—è](https://tc39.es/ecma262/multipage/abstract-operations.html#sec-tostring) –º–æ–∂–µ –±—É—Ç–∏ –æ–ø–∏—Å–∞–Ω–∞ –Ω–∞—Å—Ç—É–ø–Ω–∏–º —á–∏–Ω–æ–º:

- –†—è–¥–∫–∏ –ø–æ–≤–µ—Ä—Ç–∞—é—Ç—å—Å—è —è–∫ —î.
- [`undefined`](/uk/docs/Web/JavaScript/Reference/Global_Objects/undefined) –ø–µ—Ä–µ—Ç–≤–æ—Ä—é—î—Ç—å—Å—è –Ω–∞ `"undefined"`.
- [`null`](/uk/docs/Web/JavaScript/Reference/Operators/null) –ø–µ—Ä–µ—Ç–≤–æ—Ä—é—î—Ç—å—Å—è –Ω–∞ `"null"`.
- `true` –ø–µ—Ä–µ—Ç–≤–æ—Ä—é—î—Ç—å—Å—è –Ω–∞ `"true"`; `false` –ø–µ—Ä–µ—Ç–≤–æ—Ä—é—î—Ç—å—Å—è –Ω–∞ `"false"`.
- –ß–∏—Å–ª–∞ –ø–µ—Ä–µ—Ç–≤–æ—Ä—é—é—Ç—å—Å—è –∑–∞ —Ç–∏–º —Å–∞–º–æ –∞–ª–≥–æ—Ä–∏—Ç–º–æ–º, —â–æ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î—Ç—å—Å—è –≤ [`toString(10)`](/uk/docs/Web/JavaScript/Reference/Global_Objects/Number/toString).

- [–°–∏–º–≤–æ–ª–∏](/uk/docs/Web/JavaScript/Reference/Global_Objects/Symbol) –≤–∏–∫–∏–¥–∞—é—Ç—å {{jsxref("TypeError")}}.
- –û–±'—î–∫—Ç–∏ —Å–ø–µ—Ä—à—É [–ø–µ—Ä–µ—Ç–≤–æ—Ä—é—é—Ç—å—Å—è –Ω–∞ –ø—Ä–∏–º—ñ—Ç–∏–≤–∏](/uk/docs/Web/JavaScript/Data_structures#zvedennia-do-prymityva) —à–ª—è—Ö–æ–º –≤–∏–∫–ª–∏–∫—É —ó—Ö –º–µ—Ç–æ–¥—ñ–≤ [`[@@toPrimitive]()`](/uk/docs/Web/JavaScript/Reference/Global_Objects/Symbol/toPrimitive) (–∑—ñ `"string"` —è–∫ –ø—ñ–¥–∫–∞–∑–∫–æ—é), `toString()` —á–∏ `valueOf()` ‚Äì —É —Ç–∞–∫–æ–º—É –ø–æ—Ä—è–¥–∫—É –ø—Ä—ñ–æ—Ä–∏—Ç–µ—Ç—É. –ü—ñ—Å–ª—è —Ü—å–æ–≥–æ —Ä–µ–∑—É–ª—å—Ç—ñ–≤–Ω–∏–π –ø—Ä–∏–º—ñ—Ç–∏–≤ –ø–µ—Ä–µ—Ç–≤–æ—Ä—é—î—Ç—å—Å—è –Ω–∞ —Ä—è–¥–æ–∫.

–Ñ –¥–µ–∫—ñ–ª—å–∫–∞ —Å–ø–æ—Å–æ–±—ñ–≤ –¥–æ—Å—è–≥–Ω—É—Ç–∏ –≤ JavaScript –º–∞–π–∂–µ —Ç–∞–∫–æ–≥–æ —Å–∞–º–æ–≥–æ –µ—Ñ–µ–∫—Ç—É.

- [–®–∞–±–ª–æ–Ω–Ω–∏–π –ª—ñ—Ç–µ—Ä–∞–ª](/uk/docs/Web/JavaScript/Reference/Template_literals): `` `${x}` `` –≤–∏–∫–æ–Ω—É—î —Ç–æ—á–Ω–æ —Ç–∞–∫—ñ –∂ —Å–∞–º—ñ –∫—Ä–æ–∫–∏ –ø—Ä–∏–≤–µ–¥–µ–Ω–Ω—è, —è–∫ –æ–ø–∏—Å–∞–Ω—ñ –≤–∏—â–µ, –¥–ª—è –≤–±—É–¥–æ–≤–∞–Ω–æ–≥–æ –≤ —Å–µ–±–µ –≤–∏—Ä–∞–∑—É.

- –ó–∞—Å—Ç–æ—Å—É–≤–∞–Ω–Ω—è [–æ–ø–µ—Ä–∞—Ç–æ—Ä–∞ `+`](/uk/docs/Web/JavaScript/Reference/Operators/Addition): `"" + x` –∑–≤–æ–¥–∏—Ç—å —Å–≤—ñ–π –æ–ø–µ—Ä–∞–Ω–¥ –¥–æ _–ø—Ä–∏–º—ñ—Ç–∏–≤–∞_, –∞ –Ω–µ –ø—Ä–æ—Å—Ç–æ _—Ä—è–¥–∫–∞_, —ñ –¥–ª—è –ø–µ–≤–Ω–∏—Ö –æ–±'—î–∫—Ç—ñ–≤ –º–∞—î –≥–µ—Ç—å —ñ–Ω–∞–∫—à—É –ª–æ–≥—ñ–∫—É, –Ω—ñ–∂ –∑–≤–∏—á–∞–π–Ω–µ –ø—Ä–∏–≤–µ–¥–µ–Ω–Ω—è –¥–æ —Ä—è–¥–∫–∞. –î–∏–≤—ñ—Ç—å—Å—è –ø–æ–¥—Ä–æ–±–∏—Ü—ñ –Ω–∞ –π–æ–≥–æ [—Å—Ç–æ—Ä—ñ–Ω—Ü—ñ –¥–æ–≤—ñ–¥–∫–∏](/uk/docs/Web/JavaScript/Reference/Operators/Addition).

–ó–∞–ª–µ–∂–Ω–æ –≤—ñ–¥ –∫–æ–Ω–∫—Ä–µ—Ç–Ω–æ—ó —Å–∏—Ç—É–∞—Ü—ñ—ó –º–æ–∂–µ –º–∞—Ç–∏ –∑–º—ñ—Å—Ç –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–Ω—è `` `${x}` `` (–∞–±–∏ –∑—ñ–º—ñ—Ç—É–≤–∞—Ç–∏ –ª–æ–≥—ñ–∫—É –≤–±—É–¥–æ–≤—É–≤–∞–Ω–Ω—è) —á–∏ `String(x)` (–¥–ª—è –æ–±—Ä–æ–±–∫–∏ —Å–∏–º–≤–æ–ª—ñ–≤ –±–µ–∑ –≤–∏–∫–∏–¥–∞–Ω–Ω—è –ø–æ–º–∏–ª–æ–∫), –æ–¥–Ω–∞–∫ –Ω–µ —Å–ª—ñ–¥ –∑–∞—Å—Ç–æ—Å–æ–≤—É–≤–∞—Ç–∏ `"" + x`.

### –°–∏–º–≤–æ–ª–∏ UTF-16, –∫–æ–¥–æ–≤—ñ —Ç–æ—á–∫–∏ Unicode —Ç–∞ –≥—Ä–∞—Ñ–µ–º–Ω—ñ –∫–ª–∞—Å—Ç–µ—Ä–∏

–ü—Ä–æ—Ç–µ —É–≤–µ—Å—å –Ω–∞–±—ñ—Ä —Å–∏–º–≤–æ–ª—ñ–≤ Unicode —î –Ω–∞–±–∞–≥–∞—Ç–æ, –Ω–∞–±–∞–≥–∞—Ç–æ –±—ñ–ª—å—à–∏–º –∑–∞ 65536. –î–æ–¥–∞—Ç–∫–æ–≤—ñ —Å–∏–º–≤–æ–ª–∏ –∑–±–µ—Ä—ñ–≥–∞—é—Ç—å—Å—è –≤ UTF-16 —è–∫ _—Å—É—Ä–æ–≥–∞—Ç–Ω—ñ –ø–∞—Ä–∏_, –∫–æ—Ç—Ä—ñ —î –ø–∞—Ä–∞–º–∏ 16-–±—ñ—Ç–æ–≤–∏—Ö –∫–æ–¥–æ–≤–∏—Ö –æ–¥–∏–Ω–∏—Ü—å, –∫–æ–∂–Ω–∞ –∑ —è–∫–∏—Ö –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è—î —î–¥–∏–Ω–∏–π —Å–∏–º–≤–æ–ª. –î–ª—è —É–Ω–∏–∫–Ω–µ–Ω–Ω—è –Ω–µ–æ–¥–Ω–æ–∑–Ω–∞—á–Ω–æ—Å—Ç—ñ –¥–≤—ñ —á–∞—Å—Ç–∏–Ω–∏ –ø–∞—Ä–∏ –º—É—Å—è—Ç—å –ª–µ–∂–∞—Ç–∏ –º—ñ–∂ `0xD800` —Ç–∞ `0xDFFF`, —ñ —Ü—ñ –∫–æ–¥–æ–≤—ñ –æ–¥–∏–Ω–∏—Ü—ñ –Ω–µ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—é—Ç—å—Å—è –¥–ª—è –∫–æ–¥—É–≤–∞–Ω–Ω—è —Å–∏–º–≤–æ–ª—ñ–≤ –∑ –æ–¥–Ω—ñ—î—ó –∫–æ–¥–æ–≤–æ—ó –æ–¥–∏–Ω–∏—Ü—ñ. - –ü–µ—Ä–µ–±—É–≤–∞—î –≤ –¥—ñ–∞–ø–∞–∑–æ–Ω—ñ `0xDC00`-`0xDFFF` –≤–∫–ª—é—á–Ω–æ (—Ç–æ–±—Ç–æ —î –∫—ñ–Ω—Ü–µ–≤–∏–º —Å—É—Ä–æ–≥–∞—Ç–æ–º), –∞–ª–µ —î –ø–µ—Ä—à–æ—é –∫–æ–¥–æ–≤–æ—é –æ–¥–∏–Ω–∏—Ü–µ—é –≤ —Ä—è–¥–∫—É, –∞–±–æ –ø–æ–ø–µ—Ä–µ–¥–Ω—è –∫–æ–¥–æ–≤–∞ –æ–¥–∏–Ω–∏—Ü—è –Ω–µ —î –ø–æ—á–∞—Ç–∫–æ–≤–∏–º —Å—É—Ä–æ–≥–∞—Ç–æ–º.

–°–∞–º–æ—Ç–Ω—ñ —Å—É—Ä–æ–≥–∞—Ç–∏ –Ω–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è—é—Ç—å –∂–æ–¥–Ω–∏—Ö —Å–∏–º–≤–æ–ª—ñ–≤ Unicode. –•–æ—á–∞ –±—ñ–ª—å—à—ñ—Å—Ç—å –≤–±—É–¥–æ–≤–∞–Ω–∏—Ö –º–µ—Ç–æ–¥—ñ–≤ JavaScript –æ–±—Ä–æ–±–ª—è—î —ó—Ö –ø—Ä–∞–≤–∏–ª—å–Ω–æ, –æ—Å–∫—ñ–ª—å–∫–∏ –≤—Å—ñ –≤–æ–Ω–∏ –ø—Ä–∞—Ü—é—é—Ç—å –Ω–∞ –æ—Å–Ω–æ–≤—ñ –∫–æ–¥–æ–≤–∏—Ö –æ–¥–∏–Ω–∏—Ü—å UTF-16, —Å–∞–º–æ—Ç–Ω—ñ —Å—É—Ä–æ–≥–∞—Ç–∏ –Ω–µ—Ä—ñ–¥–∫–æ –±—É–≤–∞—é—Ç—å –Ω–µ–¥–æ–ø—É—Å—Ç–∏–º–∏–º–∏ –∑–Ω–∞—á–µ–Ω–Ω—è–º–∏ –ø—Ä–∏ –≤–∑–∞—î–º–æ–¥—ñ—ó –∑ —ñ–Ω—à–∏–º–∏ —Å–∏—Å—Ç–µ–º–∞–º–∏ ‚Äì –Ω–∞–ø—Ä–∏–∫–ª–∞–¥, [`encodeURI()`](/uk/docs/Web/JavaScript/Reference/Global_Objects/encodeURI) –≤–∏–∫–∏–¥–∞—î –Ω–∞ —Å–∞–º–æ—Ç–Ω—ñ—Ö —Å—É—Ä–æ–≥–∞—Ç–∞—Ö {{jsxref("URIError")}}, –æ—Å–∫—ñ–ª—å–∫–∏ –∫–æ–¥—É–≤–∞–Ω–Ω—è URI –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î –∫–æ–¥—É–≤–∞–Ω–Ω—è UTF-8, –∫–æ—Ç—Ä–µ –Ω–µ –º–∞—î –∫–æ–¥—É–≤–∞–Ω–Ω—è –¥–ª—è —Å–∞–º–æ—Ç–Ω—ñ—Ö —Å—É—Ä–æ–≥–∞—Ç—ñ–≤. –†—è–¥–∫–∏, —â–æ –Ω–µ –º—ñ—Å—Ç—è—Ç—å –∂–æ–¥–Ω–∏—Ö —Å–∞–º–æ—Ç–Ω—ñ—Ö —Å—É—Ä–æ–≥–∞—Ç—ñ–≤, –Ω–∞–∑–∏–≤–∞—é—Ç—å—Å—è _–¥–æ–±—Ä–µ —Å—Ñ–æ—Ä–º–æ–≤–∞–Ω–∏–º–∏_ —Ä—è–¥–∫–∞–º–∏ —ñ —î –±–µ–∑–ø–µ—á–Ω–∏–º–∏ –¥–ª—è –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–Ω—è –∑ —Ñ—É–Ω–∫—Ü—ñ—è–º–∏, –∫–æ—Ç—Ä—ñ –Ω–µ –ø—Ä–∞—Ü—é—é—Ç—å –∑ UTF-16 (—Ç–∞–∫–∏–º–∏ —è–∫ `encodeURI()` –∞–±–æ {{domxref("TextEncoder")}}). –ü–µ—Ä–µ–≤—ñ—Ä–∏—Ç–∏, —á–∏ —î —Ä—è–¥–æ–∫ –¥–æ–±—Ä–µ —Ñ–æ—Ä–º–æ–≤–∞–Ω–∏–º, –º–æ–∂–Ω–∞ –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é –º–µ—Ç–æ–¥—É {{jsxref("String/isWellFormed", "isWellFormed()")}}, –∞ —Ç–∞–∫–æ–∂ –º–æ–∂–Ω–∞ –≤–∏—á–∏—Å—Ç–∏—Ç–∏ —Å–∞–º–æ—Ç–Ω—ñ —Å—É—Ä–æ–≥–∞—Ç–∏ ‚Äì –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é –º–µ—Ç–æ–¥—É {{jsxref("String/toWellFormed", "toWellFormed()")}}.

–ù–∞ –¥–æ–¥–∞—á—É –¥–æ —Å–∏–º–≤–æ–ª—ñ–≤ Unicode, —î –ø–µ–≤–Ω—ñ –ø–æ—Å–ª—ñ–¥–æ–≤–Ω–æ—Å—Ç—ñ —Å–∏–º–≤–æ–ª—ñ–≤ Unicode, –∫–æ—Ç—Ä—ñ –ø–æ–≤–∏–Ω–Ω—ñ —Ä–æ–∑–≥–ª—è–¥–∞—Ç–∏—Å—è —è–∫ —î–¥–∏–Ω–∞ –≤–∏–¥–∏–º–∞ –æ–¥–∏–Ω–∏—Ü—è, ‚Äì –≤–æ–Ω–∏ –≤—ñ–¥–æ–º—ñ —è–∫ _–≥—Ä–∞—Ñ–µ–º–Ω—ñ –∫–ª–∞—Å—Ç–µ—Ä–∏_. –ù–∞–π–ø–æ—à–∏—Ä–µ–Ω—ñ—à–∏–π –ø—Ä–∏–∫–ª–∞–¥ ‚Äì –µ–º–æ–¥–∂—ñ: —á–∏–º–∞–ª–æ –µ–º–æ–¥–∂—ñ, –∫–æ—Ç—Ä—ñ –º–∞—é—Ç—å –∑–Ω–∞—á–Ω–∏–π –¥—ñ–∞–ø–∞–∑–æ–Ω –≤–∞—Ä—ñ–∞—Ü—ñ–π, –Ω–∞—Å–ø—Ä–∞–≤–¥—ñ —É—Ç–≤–æ—Ä—é—é—Ç—å—Å—è –¥–µ–∫—ñ–ª—å–∫–æ–º–∞ –µ–º–æ–¥–∑—ñ, –∑–∞–∑–≤–∏—á–∞–π —Å–ø–æ–ª—É—á–µ–Ω–∏–º–∏ —Å–∏–º–≤–æ–ª–æ–º \<ZWJ> (`U+200D`).
